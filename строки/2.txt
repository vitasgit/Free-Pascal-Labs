Вот несколько типичных задач для изучения строк в Free Pascal:

1. Подсчет количества слов:
```pascal
program WordCount;
var
  s: string;
  words, i: integer;
begin
  s := 'Привет мир программирование';
  words := 1;
  for i := 1 to Length(s) do
    if s[i] = ' ' then Inc(words);
  WriteLn('Количество слов: ', words);
end.
```

2. Переворот строки:
```pascal
program ReverseString;
var
  s, reversed: string;
  i: integer;
begin
  s := 'программа';
  reversed := '';
  for i := Length(s) downto 1 do
    reversed := reversed + s[i];
  WriteLn(reversed);
end.
```

3. Проверка палиндрома:
```pascal
program Palindrome;
var
  s: string;
  i: integer;
  isPalindrome: boolean;
begin
  s := 'шалаш';
  isPalindrome := true;
  for i := 1 to Length(s) div 2 do
    if s[i] <> s[Length(s)-i+1] then
    begin
      isPalindrome := false;
      break;
    end;
  if isPalindrome
    then WriteLn('Палиндром')
    else WriteLn('Не палиндром');
end.
```

4. Замена подстроки:
```pascal
program StringReplace;
var
  s: string;
begin
  s := 'привет мир';
  Delete(s, 7, 3);
  Insert('Pascal', s, 7);
  WriteLn(s);
end.
```
